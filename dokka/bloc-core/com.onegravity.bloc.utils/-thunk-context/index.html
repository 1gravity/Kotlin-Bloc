data class ThunkContext<State, Action, A : Action, Proposal>(val getState: GetState<State>, val action: A, val dispatch: Dispatcher<Action>, val reduce: (proposal: Proposal) -> Unit, runner: CoroutineRunner)

ThunkContext is used as receiver for the thunk when defining it with an action:

}

// or

}
